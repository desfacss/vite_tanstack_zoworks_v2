## Recommended: Single `commerce` Schema

Here's the clean, consolidated approach:

```sql
-- =====================================================
-- SINGLE SCHEMA: commerce (handles everything customer-facing)
-- =====================================================
CREATE SCHEMA IF NOT EXISTS commerce;

-- =====================================================
-- 1. ORDERS & TRANSACTIONS (Complete order lifecycle)
-- =====================================================
CREATE TABLE commerce.orders (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL REFERENCES identity.organizations(id),
    
    -- Customer
    customer_id uuid REFERENCES crm.contacts(id),
    email text NOT NULL,
    phone text,
    
    -- Identifiers
    order_number text NOT NULL UNIQUE,
    checkout_token text UNIQUE,
    
    -- Financials
    subtotal_price numeric(15,2) NOT NULL,
    total_discounts numeric(15,2) DEFAULT 0,
    total_tax numeric(15,2) DEFAULT 0,
    total_shipping numeric(15,2) DEFAULT 0,
    total_price numeric(15,2) NOT NULL,
    currency text DEFAULT 'INR',
    
    -- Statuses
    status text DEFAULT 'pending', -- pending, paid, processing, shipped, delivered, cancelled, refunded
    payment_status text, -- pending, paid, failed, refunded
    fulfillment_status text, -- unfulfilled, partial, fulfilled
    
    -- Addresses (JSON for flexibility)
    shipping_address jsonb,
    billing_address jsonb,
    
    -- Shipping lines
    shipping_lines jsonb, -- [{carrier, service, tracking_number, tracking_url}]
    
    -- Standards compliance (ONDC, UCP, ACP)
    external_identifiers jsonb DEFAULT '{}'::jsonb,
    -- {
    --   "ondc": {"transaction_id": "...", "provider_id": "..."},
    --   "ucp": {"session_id": "...", "checkout_id": "..."},
    --   "acp": {"conversation_id": "...", "token": "..."}
    -- }
    
    -- Metadata
    notes text,
    referrer text,
    browser_ip inet,
    
    -- Tracking
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now(),
    processed_at timestamptz,
    cancelled_at timestamptz,
    cancelled_reason text
);

CREATE TABLE commerce.order_items (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id uuid NOT NULL REFERENCES commerce.orders(id) ON DELETE CASCADE,
    organization_id uuid NOT NULL,
    
    -- Product (links to catalog)
    offering_id uuid NOT NULL REFERENCES catalog.offerings(id),
    variant_id uuid REFERENCES catalog.offering_variants(id),
    sku text,
    gtin text, -- For UCP/ONDC
    
    -- Snapshot (preserve price at purchase)
    name text NOT NULL,
    quantity numeric(10,2) NOT NULL,
    price numeric(15,2) NOT NULL,
    total_discount numeric(15,2) DEFAULT 0,
    
    -- Flags
    is_physical boolean DEFAULT true,
    is_digital boolean DEFAULT false,
    
    -- Fulfillment tracking
    fulfilled_quantity numeric(10,2) DEFAULT 0,
    
    -- Tax
    tax_lines jsonb, -- [{rate, amount}]
    
    -- Customization
    properties jsonb, -- {size: "L", color: "blue"}
    
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);

-- =====================================================
-- 2. PAYMENTS (Razorpay integration)
-- =====================================================
CREATE TABLE commerce.payments (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id uuid NOT NULL REFERENCES commerce.orders(id),
    organization_id uuid NOT NULL,
    
    -- Gateway
    gateway text NOT NULL, -- 'razorpay'
    gateway_order_id text,
    gateway_payment_id text,
    gateway_signature text,
    
    -- Amounts
    amount numeric(15,2) NOT NULL,
    currency text DEFAULT 'INR',
    
    -- Status
    status text NOT NULL, -- pending, authorized, captured, failed, refunded
    type text NOT NULL, -- payment, refund, capture, authorization
    
    -- Razorpay specifics
    razorpay_method text, -- card, upi, netbanking, wallet
    razorpay_bank_ref text,
    razorpay_error text,
    
    -- ACP delegated payment
    acp_delegated_token text,
    
    -- Metadata
    metadata jsonb,
    
    processed_at timestamptz DEFAULT now(),
    created_at timestamptz DEFAULT now()
);

-- =====================================================
-- 3. FULFILLMENT (Physical + Digital delivery)
-- =====================================================
CREATE TABLE commerce.fulfillments (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id uuid NOT NULL REFERENCES commerce.orders(id),
    organization_id uuid NOT NULL,
    
    -- Tracking
    tracking_number text,
    carrier text,
    tracking_url text,
    
    -- Location
    location_id uuid REFERENCES identity.locations(id),
    
    -- Status
    status text NOT NULL, -- pending, shipped, delivered, failed, returned
    shipped_at timestamptz,
    delivered_at timestamptz,
    
    -- Items in this shipment
    items jsonb NOT NULL, -- [{order_item_id, quantity}]
    
    -- Digital delivery (for digital products)
    digital_assets jsonb, -- [{url, download_key, expires_at, file_name, file_size}]
    
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);

-- =====================================================
-- 4. CARTS (Session management)
-- =====================================================
CREATE TABLE commerce.carts (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL,
    
    -- Customer (guest or logged in)
    customer_id uuid REFERENCES crm.contacts(id),
    session_token text NOT NULL UNIQUE,
    
    -- Cart contents
    items jsonb NOT NULL DEFAULT '[]'::jsonb,
    -- [
    --   {
    --     "offering_id": "uuid",
    --     "variant_id": "uuid", 
    --     "quantity": 2,
    --     "properties": {"size": "L"}
    --   }
    -- ]
    
    -- Discounts
    discount_codes text[],
    discount_applications jsonb,
    
    -- Expiry
    expires_at timestamptz NOT NULL DEFAULT (now() + interval '7 days'),
    converted_to_order_id uuid REFERENCES commerce.orders(id),
    
    -- ONDC cart persistence
    external_cart_id text,
    
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);

-- =====================================================
-- 5. RETURNS & REFUNDS
-- =====================================================
CREATE TABLE commerce.returns (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    order_id uuid NOT NULL REFERENCES commerce.orders(id),
    organization_id uuid NOT NULL,
    
    -- Return details
    return_number text NOT NULL UNIQUE,
    reason text,
    status text NOT NULL, -- requested, approved, received, refunded, rejected
    
    -- Items returned
    items jsonb NOT NULL, -- [{order_item_id, quantity, reason}]
    
    -- Refund
    refund_amount numeric(15,2),
    refund_payment_id uuid REFERENCES commerce.payments(id),
    
    -- Tracking (customer return shipment)
    return_tracking_number text,
    return_carrier text,
    
    -- Metadata
    notes text,
    
    requested_at timestamptz DEFAULT now(),
    approved_at timestamptz,
    received_at timestamptz,
    refunded_at timestamptz,
    
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);

-- =====================================================
-- 6. REVIEWS & RATINGS (Optional but valuable)
-- =====================================================
CREATE TABLE commerce.reviews (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
    organization_id uuid NOT NULL,
    offering_id uuid NOT NULL REFERENCES catalog.offerings(id),
    order_id uuid REFERENCES commerce.orders(id),
    customer_id uuid REFERENCES crm.contacts(id),
    
    rating integer NOT NULL CHECK (rating BETWEEN 1 AND 5),
    title text,
    content text,
    is_verified_purchase boolean DEFAULT false,
    status text DEFAULT 'pending', -- pending, approved, rejected, flagged
    
    metadata jsonb, -- {images: ["url1", "url2"], variant_id: "uuid"}
    
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);
```

## Why This Structure Works Better

### Single Schema Benefits:
1. **Cleaner joins** - All commerce-related data in one place
2. **Easier maintenance** - One schema, one set of migrations
3. **Simpler RLS policies** - One set of tenant isolation rules
4. **Faster development** - No schema hopping in queries

### Clear Separation of Concerns:

| Schema | Responsibility |
|--------|---------------|
| **catalog** | What you sell (products, variants, prices, inventory) |
| **commerce** | How you sell (orders, payments, fulfillment, carts) |
| **finance** | Accounting (invoices, bills, ledgers) |
| **crm** | Who you sell to (contacts, customers) |
| **identity** | Multi-tenant (orgs, users, locations) |

## Integration Points

### Bridge from catalog to commerce:
```sql
-- Auto-deduct inventory when order is paid
CREATE OR REPLACE FUNCTION commerce.deduct_inventory()
RETURNS TRIGGER AS $$
BEGIN
    -- Update inventory levels for physical products
    UPDATE catalog.inventory_levels il
    SET quantity = quantity - oi.quantity,
        updated_at = now()
    FROM commerce.order_items oi
    WHERE oi.order_id = NEW.order_id
      AND oi.is_physical = true
      AND il.offering_variant_id = oi.variant_id;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_deduct_inventory
    AFTER UPDATE OF payment_status ON commerce.orders
    FOR EACH ROW
    WHEN (NEW.payment_status = 'paid')
    EXECUTE FUNCTION commerce.deduct_inventory();
```

### Bridge from commerce to finance:
```sql
-- Auto-create invoice when order is paid
CREATE OR REPLACE FUNCTION commerce.create_invoice()
RETURNS TRIGGER AS $$
DECLARE
    invoice_id uuid;
    invoice_number text;
BEGIN
    -- Generate invoice number (e.g., INV-2024-00001)
    invoice_number := 'INV-' || to_char(NEW.created_at, 'YYYY') || '-' || 
                      lpad(NEW.order_number::text, 5, '0');
    
    -- Create invoice in finance schema
    INSERT INTO finance.invoices (
        organization_id, contact_id, invoice_number, 
        total_amount, due_date, details, status
    ) VALUES (
        NEW.organization_id, 
        NEW.customer_id,
        invoice_number,
        NEW.total_price,
        NEW.created_at + interval '30 days', -- Net 30 terms
        (SELECT jsonb_agg(jsonb_build_object(
            'description', name,
            'quantity', quantity,
            'unit_price', price,
            'total', price * quantity
        )) FROM commerce.order_items WHERE order_id = NEW.id),
        'paid'
    );
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trigger_create_invoice
    AFTER UPDATE OF payment_status ON commerce.orders
    FOR EACH ROW
    WHEN (NEW.payment_status = 'paid')
    EXECUTE FUNCTION commerce.create_invoice();
```
